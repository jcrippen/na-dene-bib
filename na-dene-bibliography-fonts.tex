%%%
%%% Fonts and symbols.
%%%

%% XeLaTeX font support.
\usepackage{fontspec}
\usepackage{xltxtra}

%% The base font from which most other fonts are derived.
\newcommand{\basefont}{Brill}

%% The essential font definitions.
\setmainfont{\basefont}
\defaultfontfeatures{Scale=MatchLowercase}
\setsansfont[LetterSpace=-0.1]{Calibri}
\setmonofont[FakeStretch=0.9]{Consolas}

%% Page headers and footers.
\newfontfamily\pageheadfont{\basefont}
\newfontfamily\pagefootfont{\basefont}

%%
%% Special fonts.
%%

%% Some of these font definitions are used by stuff in the
%% ...-fonts-symbols.tex file that gets loaded after
%% this file.

%% Phonetics.
%\newfontfamily{\ipafont}{Charis SIL}
%\newfontfamily{\ipafont}{Doulos SIL}
\newfontfamily{\ipafont}{\basefont}
\newfontfamily{\ipaspecialfont}{Doulos SIL}

%% This font is for unusual phonetic symbols.
\newfontfamily{\phonfont}{Charis SIL Compact}

%%
%% Don't call it \mathfont because other packages use that name.
%% You have been warned.
%\newfontfamily{\mthfont}{Cambria Math}
